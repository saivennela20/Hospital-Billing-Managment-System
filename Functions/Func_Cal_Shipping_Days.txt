create or replace FUNCTION FUNC_CAL_SHIPPING_DAYS(P_DELIVERY_MODE VARCHAR2)
RETURN NUMBER
AS
V_SHIPPING_DAYS NUMBER;
BEGIN
CASE
WHEN P_DELIVERY_MODE = 'STANDARD'
THEN V_SHIPPING_DAYS := 3;
WHEN P_DELIVERY_MODE = 'EXPEDITED'
THEN V_SHIPPING_DAYS := 1;
END CASE;
RETURN V_SHIPPING_DAYS;
END;
